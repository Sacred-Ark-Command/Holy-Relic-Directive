      const statGrades = ['D', 'C', 'B', 'A', 'S'];

      function generateStats() {
        // 1. 定义稀有度及其权重
        // S: 1/20 = 5%
        // A: 3/20 = 15%
        // B: 6/20 = 30%
        // C: 10/20 = 50%
        const rarityConfig = [
          { name: 'C', weight: 10 },
          { name: 'B', weight: 6 },
          { name: 'A', weight: 3 },
          { name: 'S', weight: 1 }, // 1 in 20
        ];

        // 辅助函数：根据权重随机选择
        const getWeightedRandom = config => {
          const totalWeight = config.reduce((sum, item) => sum + item.weight, 0);
          let random = Math.random() * totalWeight;
          for (const item of config) {
            if (random < item.weight) {
              return item.name;
            }
            random -= item.weight;
          }
        };

        // 2. 生成稀有度
        const generatedRarity = getWeightedRandom(rarityConfig);

        // 3. 定义所有可能的属性等级
        const allGrades = ['D', 'C', 'B', 'A', 'S'];
        let stats = [];

        // 辅助函数：从池中随机获取一个等级
        const getRandomGrade = pool => pool[Math.floor(Math.random() * pool.length)];

        // 4. 根据稀有度生成属性
        switch (generatedRarity) {
          case 'C':
            // C级的五维最高为B级
            const cPool = allGrades.slice(0, 3); // -> ['D', 'C', 'B']
            stats = Array(5)
              .fill(null)
              .map(() => getRandomGrade(cPool));
            break;
          case 'B':
            // B级的五维最高为A级
            const bPool = allGrades.slice(0, 4); // -> ['D', 'C', 'B', 'A']
            stats = Array(5)
              .fill(null)
              .map(() => getRandomGrade(bPool));
            break;
          case 'A':
            // A级的五维最高为S但有且仅有一个
            const aPool = allGrades.slice(0, 4); // -> ['D', 'C', 'B', 'A']
            stats = Array(4)
              .fill(null)
              .map(() => getRandomGrade(aPool));
            stats.push('S');
            // 洗牌算法打乱数组，确保'S'的位置是随机的
            for (let i = stats.length - 1; i > 0; i--) {
              const j = Math.floor(Math.random() * (i + 1));
              [stats[i], stats[j]] = [stats[j], stats[i]];
            }
            break;
          case 'S':
            // S级的五维最低为B级
            const sPool = allGrades.slice(2); // -> ['B', 'A', 'S']
            stats = Array(5)
              .fill(null)
              .map(() => getRandomGrade(sPool));
            break;
        }

        // 5. 更新界面
        document.getElementById('rarity-val').textContent = generatedRarity;
        document.getElementById('stat1-val').textContent = stats[0];
        document.getElementById('stat2-val').textContent = stats[1];
        document.getElementById('stat3-val').textContent = stats[2];
        document.getElementById('stat4-val').textContent = stats[3];
        document.getElementById('stat5-val').textContent = stats[4];
      }
