      function GetRawValue(obj, path, defaultValue = null) {
        const keys = Array.isArray(path) ? path : path.split('.');
        let current = obj;
        for (let i = 0; i < keys.length; i++) {
          const key = keys[i];
          if (
            current === undefined ||
            current === null ||
            typeof current !== 'object' ||
            !current.hasOwnProperty(key)
          ) {
            return defaultValue;
          }
          current = current[key];
        }
        return current === undefined || current === null ? defaultValue : current;
      }

      function SafeGetValue(obj, path, defaultValue = 'N/A') {
        // Path can be a string 'a.b.c' or an array ['a', 'b', 'c']
        const keys = Array.isArray(path) ? path : path.split('.');
        let current = obj;
        for (let i = 0; i < keys.length; i++) {
          const key = keys[i];
          if (
            current === undefined ||
            current === null ||
            typeof current !== 'object' ||
            !current.hasOwnProperty(key)
          ) {
            // console.warn(`Path not found: ${path} at key: ${key}`);
            return defaultValue;
          }
          current = current[key];
        }

        if (current === undefined || current === null) {
          return defaultValue;
        }

        // If the value is an array, return the first element (the actual value).
        if (Array.isArray(current) && current.length > 0) {
          const actualValue = current[0];
          return typeof actualValue === 'boolean' ? actualValue : String(actualValue);
        }

        return typeof current === 'boolean' ? current : String(current);
      }

      function updateProgressBar(barId, valueMainId, valueDescId, rawValue, min, max, stageMapping) {
        const progressBar = document.getElementById(barId);
        const valueMainDisplay = document.getElementById(valueMainId);
        const valueDescDisplay = document.getElementById(valueDescId);

        if (!valueMainDisplay) return;

        const numericValue = parseFloat(rawValue) || min;
        valueMainDisplay.innerText = String(rawValue);

        if (valueDescDisplay && stageMapping) {
          const stage = stageMapping.find(s => numericValue >= s.min && numericValue <= s.max);
          valueDescDisplay.innerText = stage ? stage.name : '';
        }

        if (progressBar) {
          const clampedValue = Math.max(min, Math.min(max, numericValue));
          const range = max - min;
          const percentage = range === 0 ? (clampedValue >= max ? 100 : 0) : ((clampedValue - min) / range) * 100;
          progressBar.style.width = `${percentage}%`;

          if (stageMapping) {
            const stage = stageMapping.find(s => clampedValue >= s.min && clampedValue <= s.max);
            if (stage && stage.color) {
              progressBar.style.backgroundColor = stage.color;
              // No box-shadow for wasteland theme
            }
          }
        }
      }

      function openUpgradeModal() {
        if (!statDataCache) return;
        const modal = document.getElementById('upgrade-modal');

        resourceList.innerHTML = '';
        const resourceSkipKeys = ['$meta', '武备库', '物品库'];
        for (const resourceName in currentResources) {
            if (!currentResources.hasOwnProperty(resourceName) || resourceSkipKeys.includes(resourceName)) continue;
            const rawVal = GetRawValue(currentResources, resourceName);
            // 跳过非简单值（对象/数组）
            if (typeof rawVal === 'object' && rawVal !== null) continue;
            const value = SafeGetValue(currentResources, resourceName, '0');
            resourceList.innerHTML += `<div>${resourceName}: <span style="color: var(--text-primary);">${value}</span></div>`;

        const upgradeLevelData = GetRawValue(statDataCache, '陆行舰.等级数据', {});

      function showUpgradeConfirmation(facilityName, currentLevel) {
        const dialog = document.getElementById('upgrade-confirmation-dialog');
        const text = document.getElementById('upgrade-confirmation-text');
        const costList = document.getElementById('upgrade-cost-list');
        const warning = document.getElementById('upgrade-resource-warning');
        const confirmBtn = document.getElementById('upgrade-confirm-execute-btn');

        const nextLevel = currentLevel + 1;
        text.innerText = `确定要将 ${facilityName} 升级到 Lv.${nextLevel} 吗?`;

        const costs = upgradeCosts[facilityName]?.[nextLevel];
        if (!costs) {
            costList.innerHTML = '<span>无法获取升级成本信息。</span>';
            confirmBtn.disabled = true;
            dialog.style.display = 'block';
            return;
        }

        const currentResources = GetRawValue(statDataCache, '陆行舰.资源', {});
        let hasEnoughResources = true;
        let costHtml = '<strong style="margin-bottom: 5px; display: block;">资源消耗:</strong>';
        costList.style.display = 'grid';
        costList.style.gridTemplateColumns = '1fr 1fr';
        costList.style.gap = '5px 10px';


        for (const resource in costs) {
            const required = costs[resource];
            const current = parseInt(SafeGetValue(currentResources, resource, '0'), 10);
            const color = current >= required ? 'var(--text-primary)' : 'var(--accent-danger)';
            costHtml += `<div class="upgrade-cost" style="color: ${color};">${resource}: ${required} (拥有: ${current})</div>`;
            if (current < required) {
                hasEnoughResources = false;
            }
        }
        costList.innerHTML = costHtml;

        if (hasEnoughResources) {
            warning.innerText = '';
            confirmBtn.disabled = false;
        } else {
            warning.innerText = '资源不足!';
            confirmBtn.disabled = true;
        }

        confirmBtn.onclick = () => executeUpgrade(facilityName, nextLevel, costs);

        // Hide the main facility list and show the confirmation dialog
        document.getElementById('upgrade-facility-list').style.display = 'none';
        dialog.style.display = 'block';
      }

            alert('无法执行升级操作。');

      function createExpandableItemSection(title, sectionData, itemPathPrefix, statusGrid, insertBeforeElement = null) {
        // Do not render if data is missing or empty (ignoring meta)
        // 新变量结构中，key 本身就是名称，内部没有 '名称' 字段
        if (!sectionData || typeof sectionData !== 'object' || Object.keys(sectionData).filter(k => k !== '$meta').length === 0) {
            return;
        }

        const sectionEl = document.createElement('div');
        sectionEl.className = 'section dynamic-ship-section';
        sectionEl.innerHTML = `
            <div class="section-header clickable" onclick="toggleSection(this)">${title}</div>
            <div class="section-content" style="display: none; gap: 10px;"></div>
        `;
        const contentEl = sectionEl.querySelector('.section-content');

        for (const itemKey in sectionData) {
            if (!sectionData.hasOwnProperty(itemKey) || itemKey === '$meta') continue;

            const item = sectionData[itemKey];
            const itemName = itemKey;
            const itemStatus = SafeGetValue(item, '状态', '未知');
            const itemLevel = SafeGetValue(item, '等级', '');
            const itemType = SafeGetValue(item, '类型', '');

            // 构建效果文本
            const itemEffects = GetRawValue(item, '效果', {});
            const effectParts = [];
            for (const ek in itemEffects) {
              if (itemEffects.hasOwnProperty(ek) && ek !== '$meta') effectParts.push(ek + ': ' + itemEffects[ek]);
            }
            const effectText = effectParts.length > 0 ? effectParts.join('；') : '';

            // 构建装备子列表（机库载具有装备）
            const subEquipData = GetRawValue(item, '装备', {});
            const subEquipKeys = Object.keys(subEquipData).filter(k => k !== '$meta');
            let subEquipHtml = '';
            if (subEquipKeys.length > 0) {
              subEquipHtml = '<div style="margin-top: 6px; padding-left: 8px; border-left: 2px solid var(--border-color);">';
              subEquipKeys.forEach(seKey => {
                const se = subEquipData[seKey];
                const seLevel = SafeGetValue(se, '等级', 'N/A');
                const seType = SafeGetValue(se, '类型', 'N/A');
                const sePos = SafeGetValue(se, '位置', '');
                subEquipHtml += `<div style="margin-bottom: 4px;"><span style="color: var(--text-header);">${seKey}</span> <span class="value-description">${seLevel} [${seType}]${sePos ? ' 位置:' + sePos : ''}</span></div>`;
              });
              subEquipHtml += '</div>';
            }

            // 标题行附加信息
            const levelInfo = itemLevel ? ` Lv.${itemLevel}` : '';
            const typeInfo = itemType ? ` [${itemType}]` : '';

            const itemEl = document.createElement('div');
            itemEl.className = 'property';
            itemEl.style.cursor = 'pointer';
            const itemPath = `${itemPathPrefix}.${itemKey}`;
            itemEl.setAttribute('onclick', `toggleItemDescription(this, '${itemPath}')`);
            itemEl.innerHTML = `
                <div class="property-name" style="text-transform: none;">${itemName}<span class="value-description" style="margin-left: 8px;">${levelInfo}${typeInfo}</span></div>
                <div class="property-value-container">
                    <span class="value-main" style="font-size: 1em; color: ${itemStatus === '良好' ? 'var(--accent-bond)' : 'var(--accent-warning)'};">${itemStatus}</span>
                </div>
                <div class="item-description">
                  ${effectText ? `<div style="margin-bottom: 4px;">效果: ${effectText}</div>` : ''}
                  ${subEquipHtml}
                </div>
            `;
            contentEl.appendChild(itemEl);
        }

        // Only append the section if it has actual items to show
        if (contentEl.children.length > 0) {
            if (insertBeforeElement) {
                statusGrid.insertBefore(sectionEl, insertBeforeElement);
            } else {
                statusGrid.appendChild(sectionEl);
            }
        }
      }

      // 辅助函数：将人形列表对象转换为数组（新变量结构中人形列表是对象，key是代号）
      function getDollListAsArray(statData) {
        const dollObj = GetRawValue(statData, '人形列表', {});
        if (!dollObj || typeof dollObj !== 'object') return [];
        const result = [];
        for (const code in dollObj) {
          if (!dollObj.hasOwnProperty(code) || code === '$meta') continue;
          let dollData = dollObj[code];
          if (dollData && typeof dollData === 'object') {
            dollData._代号 = code; // 注入代号作为内部标识
            result.push(dollData);
          }
        }
        return result;
      }

      function sortDolls(key, event) {
        event.stopPropagation(); // Prevent the section from toggling

        if (currentSort.key === key) {
            currentSort.direction = currentSort.direction === 'asc' ? 'desc' : 'asc';
        } else {
            currentSort.key = key;
            currentSort.direction = 'desc'; // Default to descending for new sort key
        }

        let dollList = getDollListAsArray(statDataCache);
        if (!Array.isArray(dollList) || dollList.length === 0) return;

        const rarityOrder = { 'C': 0, 'B': 1, 'A': 2, 'S': 3, 'SS': 4, 'SSS': 5 };

        dollList.sort((a, b) => {
            let valA, valB;
            switch (key) {
                case 'rarity':
                    valA = rarityOrder[SafeGetValue(a, '稀有度', 'C')] || 0;
                    valB = rarityOrder[SafeGetValue(b, '稀有度', 'C')] || 0;
                    break;
                case 'bond':
                    valA = parseInt(SafeGetValue(a, '好感', '0'), 10);
                    valB = parseInt(SafeGetValue(b, '好感', '0'), 10);
                    break;
                default:
                    return 0;
            }

            if (currentSort.direction === 'asc') {
                return valA - valB;
            } else {
                return valB - valA;
            }
        });

        renderDollList(dollList);
      }

      function toggleSection(headerElement) {
        // Stop propagation if a sort button was clicked
        if (event.target.classList.contains('sort-btn')) {
            event.stopPropagation();
            return;
        }
        const content = headerElement.nextElementSibling;
        if (content) {
          const isVisible = content.style.display === 'flex' || content.style.display === 'block';
          content.style.display = isVisible ? 'none' : 'flex';
        }
      }

      // --- Confirmation Modal Logic ---
      function showConfirmationModal(title, text, onConfirmCallback) {
        const modal = document.getElementById('confirmation-modal');
        if (!modal) return;

        document.getElementById('confirmation-modal-title').innerText = title;
        document.getElementById('confirmation-modal-text').innerText = text;

        const confirmBtn = document.getElementById('confirmation-modal-confirm-btn');

        // Clone and replace the button to remove old event listeners
        const newConfirmBtn = confirmBtn.cloneNode(true);
        confirmBtn.parentNode.replaceChild(newConfirmBtn, confirmBtn);

        newConfirmBtn.onclick = () => {
          onConfirmCallback();
          closeConfirmationModal();
        };

        modal.style.display = 'block';
      }

        const modal = document.getElementById('confirmation-modal');
        if (modal) {
          modal.style.display = 'none';
        }
      }
