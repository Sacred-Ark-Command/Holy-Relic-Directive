    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const dataInput = document.getElementById('data-input');
            const dollListContainer = document.getElementById('doll-list-container');
            const listPlaceholder = document.getElementById('list-placeholder');
            const curtainWrapper = document.getElementById('curtain-wrapper');

            const rankStyles = {
                'S': { color: 'var(--rank-S)', name: 'S' },
                'A': { color: 'var(--rank-A)', name: 'A' },
                'B': { color: 'var(--rank-B)', name: 'B' },
                'C': { color: 'var(--rank-C)', name: 'C' },
                'D': { color: 'var(--rank-D)', name: 'D' }
            };

            function initialize(rawData) {
                const allDolls = parseDolls(rawData);

                // Data is ready, wait for user interaction to reveal
                curtainWrapper.addEventListener('click', () => {
                    curtainWrapper.classList.add('reveal');
                    curtainWrapper.classList.add('revealed'); // Disable pointer events after revealing

                    if (allDolls.length > 0) {
                        setTimeout(() => {
                            listPlaceholder.style.display = 'none';
                            displayDollCards(allDolls);
                        }, 500);
                    } else {
                        showError();
                    }
                }, { once: true }); // Ensure the event only fires once
            }

            function parseDolls(data) {
                const parsedDolls = [];
                const dollRegex = /<doll>([\s\S]*?)<\/doll>/g;
                const tagRegex = (tagName) => new RegExp(`<${tagName}>([\\s\\S]*?)<\\/${tagName}>`);

                let dollMatch;
                while ((dollMatch = dollRegex.exec(data)) !== null) {
                    const dollContent = dollMatch[1];
                    const getTagValue = (tagName) => {
                        const match = dollContent.match(tagRegex(tagName));
                        return match ? match[1].trim() : 'N/A';
                    };

                    parsedDolls.push({
                        id: `doll-${parsedDolls.length}`,
                        code: getTagValue('code'),
                        rare: getTagValue('rare'),
                        looks: getTagValue('looks'),
                        wear: getTagValue('wear'),
                        per: getTagValue('per'),
                        war: getTagValue('war'),
                        forte: getTagValue('forte'),
                        armed: getTagValue('armed'),
                        T: getTagValue('T'),
                        D: getTagValue('D'),
                        M: getTagValue('M'),
                        E: getTagValue('E'),
                        G: getTagValue('G'),
                    });
                }
                return parsedDolls;
            }

            function displayDollCards(dolls) {
                dollListContainer.innerHTML = '';
                dolls.forEach(doll => {
                    const card = document.createElement('div');
                    card.className = 'doll-card';
                    card.id = doll.id;

                    const rarity = doll.rare.toUpperCase();
                    const rarityInfo = rankStyles[rarity] || { color: 'var(--text-secondary)', name: rarity };

                    // Function to convert hex to rgb values for the glow
                    const hexToRgb = (hex) => {
                        let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
                        return result ? `${parseInt(result[1], 16)}, ${parseInt(result[2], 16)}, ${parseInt(result[3], 16)}` : '255, 255, 255';
                    };

                    card.style.setProperty('--glow-color', rarityInfo.color);
                    card.style.setProperty('--glow-rgb', hexToRgb(rarityInfo.color));
                    card.style.borderLeftColor = rarityInfo.color;

                    card.innerHTML = `
                        <div class="doll-card-summary">
                            <div class="doll-card-code">${doll.code}</div>
                            <div class="doll-card-rare" style="color: ${rarityInfo.color};">${rarityInfo.name}</div>
                        </div>
                        <div class="detail-content"></div>
                    `;

                    card.addEventListener('click', () => {
                        const wasActive = card.classList.contains('active');

                        // Close all cards
                        document.querySelectorAll('.doll-card').forEach(c => {
                            c.classList.remove('active');
                            const content = c.querySelector('.detail-content');
                            if (content) content.innerHTML = ''; // Clear content to save memory
                        });

                        // If it wasn't active, open it
                        if (!wasActive) {
                            card.classList.add('active');
                            const detailContent = card.querySelector('.detail-content');
                            if (detailContent) {
                                displayDollDetail(doll, detailContent);
                            }
                        }
                    });
                    dollListContainer.appendChild(card);
                });
            }

            function displayDollDetail(doll, container) {
                const rarity = doll.rare.toUpperCase();
                const rarityInfo = rankStyles[rarity] || { color: 'var(--text-secondary)', name: rarity };

                const getStatHTML = (label, value) => {
                    const statRank = value.toUpperCase();
                    const statInfo = rankStyles[statRank] || { color: 'var(--text-secondary)' };
                    return `<div class="stat-item"><div class="stat-label">${label}</div><div class="stat-value" style="color: ${statInfo.color};">${value}</div></div>`;
                };

                container.innerHTML = `
                    <div class="stats-grid">
                        ${getStatHTML('战术执行力', doll.T)}
                        ${getStatHTML('素体韧性', doll.D)}
                        ${getStatHTML('心智稳定度', doll.M)}
                        ${getStatHTML('环境适应性', doll.E)}
                        ${getStatHTML('成长潜质', doll.G)}
                    </div>
                    <div class="detail-section"><div class="detail-label">外貌</div><div class="detail-value">${doll.looks}</div></div>
                    <div class="detail-section"><div class="detail-label">着装</div><div class="detail-value">${doll.wear}</div></div>
                    <div class="detail-section"><div class="detail-label">性格</div><div class="detail-value">${doll.per}</div></div>
                    <div class="detail-section"><div class="detail-label">职业</div><div class="detail-value">${doll.war}</div></div>
                    <div class="detail-section"><div class="detail-label">专长</div><div class="detail-value">${doll.forte}</div></div>
                    <div class="detail-section"><div class="detail-label">人形武装</div><div class="detail-value">${doll.armed}</div></div>
                `;
            }

            function showError() {
                listPlaceholder.innerHTML = '<p>数据接收失败或格式错误。<br>请检查输入源。</p>';
            }

            function decodeHtml(html) {
                var txt = document.createElement("textarea");
                txt.innerHTML = html;
                return txt.value;
            }

            const interval = 50, timeout = 2000; let elapsedTime = 0;
            const dataObserver = setInterval(() => {
                const rawData = dataInput.value;
                elapsedTime += interval;
                if (rawData && rawData.trim() !== '$' + '1') {
                    clearInterval(dataObserver);
                    const decodedData = decodeHtml(rawData);
                    initialize(decodedData);
                } else if (elapsedTime >= timeout) {
                    clearInterval(dataObserver);
                    showError();
                }
            }, interval);
        });
    </script>
